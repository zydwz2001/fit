<script>
    let selectedFolderId = null;

    function nav(page, el) {
        document.querySelectorAll('.sub-page').forEach(p => p.classList.add('hidden'));
        document.querySelectorAll('.nav-item').forEach(i => i.classList.remove('active'));
        el.classList.add('active');
        if (page === 'training') {
            document.getElementById('training-header').classList.remove('hidden');
            showSubPage('today');
        } else {
            document.getElementById('training-header').classList.add('hidden');
            document.getElementById('page-' + page).classList.remove('hidden');
        }
    }

    function showSubPage(id) {
        document.querySelectorAll('.sub-page').forEach(p => p.classList.add('hidden'));
        document.querySelectorAll('.sub-tab-btn').forEach(b => b.classList.remove('active'));
        document.getElementById('page-' + id).classList.remove('hidden');
        document.getElementById('btn-' + id).classList.add('active');
    }

    // 身体模块逻辑
    function switchMetric(type) {
        document.querySelectorAll('.metric-card').forEach(c => c.classList.remove('active'));
        event.currentTarget.classList.add('active');
    }

    function updateTarget(type, val) {
        if(type === 'weight') {
            document.getElementById('target-text').textContent = '目标: ' + val;
            // 简单模拟折线图虚线高度变动
            const y = 30 - (parseFloat(val) - 50) * 2; 
            document.getElementById('target-line').setAttribute('y1', y);
            document.getElementById('target-line').setAttribute('y2', y);
            document.getElementById('target-text').setAttribute('y', y-1);
        }
    }

    // 知识模块逻辑
    function toggleFolder(id, e) {
        e.stopPropagation();
        const content = document.getElementById(id);
        const icon = document.getElementById(id.replace('content', 'icon'));
        content.classList.toggle('open');
        icon.style.transform = content.classList.contains('open') ? 'rotate(180deg)' : 'rotate(0deg)';
    }

    function selectFolder(id) {
        document.querySelectorAll('.folder-item').forEach(f => {
            f.classList.remove('border-emerald-500', 'ring-1', 'ring-emerald-500');
        });
        const active = document.getElementById(id);
        active.classList.add('border-emerald-500', 'ring-1', 'ring-emerald-500');
        selectedFolderId = id;
    }

    function showKnowledgeActions() {
        const type = selectedFolderId ? "笔记" : "文件夹";
        const msg = selectedFolderId ? `已选择文件夹，新增笔记将存入其中` : `未选择文件夹，将新增根目录文件夹`;
        console.log(msg);
        // 此处可接入创建逻辑
    }
</script>
